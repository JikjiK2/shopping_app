#include <iostream>
#include <string>
#include <math.h>
#include <vector>
#include <stdlib.h>
#include <stack>
#include <algorithm>
#include<fstream>

using namespace std;

#define path "cities30.txt" // 파일 경로
#define pointNum 30 //좌표 개수
#define mutation 1 //돌연변이
#define gene //세대수

class Point {
public:
	int x;
	int y;
};

Point arr[pointNum];

void File_Read() {	
		
	FILE* fp;

	fopen_s(&fp, path, "r+");

	//좌표 개수 알 때

	for (int i = 0; i < pointNum; i++) {
		int tempArr[2];
		fscanf_s(fp, "%d %d", &tempArr[0], &tempArr[1]);
		arr[i].x = tempArr[0];
		arr[i].y = tempArr[1];
	}
	fclose(fp);
}

double DisSum(Point arr[]) {
	double sum = 0;
	for (int i = 0; i < pointNum - 1; i++) {
		sum += sqrt(pow(arr[i].x - arr[i + 1].x, 2) + pow(arr[i].y - arr[i + 1].y, 2));
	}
	return sum;
}

int main(){

	File_Read();

	cout << DisSum(arr) << endl;

	return 0;
}
